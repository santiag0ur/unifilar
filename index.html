        let currentIndex = 0, score = 0, shuffled = [];


        function init() {

            shuffled = [...database].sort(() => Math.random() - 0.5);

            render();

        }


        function render() {

            const q = shuffled[currentIndex];

            document.getElementById("current-idx").innerText = currentIndex + 1;

            document.getElementById("image-box").innerHTML = `<img src="img/${q.img}" class="quiz-img">`;

            

            let options = [q.ans];

            while(options.length < 4) {

                let r = database[Math.floor(Math.random()*database.length)].ans;

                if(!options.includes(r)) options.push(r);

            }

            options.sort(() => Math.random() - 0.5);


            const box = document.getElementById("options-box");

            box.innerHTML = "";

            options.forEach(opt => {

                const b = document.createElement("button");

                b.innerText = opt;

                b.onclick = () => {

                    const all = box.querySelectorAll("button");

                    all.forEach(btn => {

                        btn.disabled = true;

                        if(btn.innerText === q.ans) btn.classList.add("correct");

                    });

                    if(opt === q.ans) { score++; } else { b.classList.add("incorrect"); }

                    document.getElementById("next-btn").style.display = "inline-block";

                };

                box.appendChild(b);

            });

            document.getElementById("next-btn").style.display = "none";

        }
